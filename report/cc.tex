      \left[
          \begin{array}[h]{c|c}
            u_i & A u_i
          \end{array}
        \right]
      \quad \text{where} \quad  \begin{dcases}
        H = \left[
          \begin{array}[h]{c|c}
            B & C
          \end{array}
        \right] \\
       A = C^{-1} B
      \end{dcases}
    \end{equation*}

      \begin{split}
        d_i
        = MOD(c_i)
        = & \begin{dcases}
          +1 & c_i = 1 \\
          -1 & c_i = 0
        \end{dcases} \\
        r_i
        = d_i + w_i \quad\text{where}\quad
        & \begin{dcases}
          w_i \sim \mathcal{N}(0, \sigma^2) \\
          \sigma^2 = \left( 2R~ \frac{E_b}{N_o} \right)^{-1}
        \end{dcases}
      \end{split}
    \end{equation*}
